```python
from rich.console import Console
from rich.text import Text


def display_prompt(
    console: Console,
    prompt: InteractivePrompt,
    *,
    index: int,
    total: int,
    filename: str,
    source: str | None = None,
    companions: list | None = None,
) -> None:
    """Render the interactive prompt to the console."""
    # Header: [N/total] filename
    header = Text()
    header.append(f"[{index}/{total}] ", style="bold cyan")
    header.append(filename)
    console.print(header)

    candidates = prompt.candidates
    if not candidates:
        console.print("      -> no match found", style="yellow")
    elif len(candidates) == 1 or (
        len(candidates) > 1
        and prompt.default_action == PromptAction.ACCEPT
    ):
        c = candidates[0]
        year_str = f" ({c.year})" if c.year else ""
        conf_str = f"  {c.confidence:.0%}" if c.confidence < 1.0 else ""
        source_str = f"  [{source}]" if source else ""
        console.print(
            f"      -> {c.title}{year_str}  tmdb:{c.tmdb_id}{conf_str}{source_str}"
        )
    else:
        console.print("      -> multiple candidates:")
        for i, c in enumerate(candidates, 1):
            year_str = f" ({c.year})" if c.year else ""
            conf_str = f"  {c.confidence:.0%}"
            source_label = f"  [{source}]" if source and i == 1 else ""
            console.print(
                f"        {i}. {c.title}{year_str}  tmdb:{c.tmdb_id}{conf_str}{source_label}"
            )

    # Companion file list
    if companions:
        console.print()
        for comp in companions:
            check = "+" if comp.move_by_default else "?"
            cat_str = f"{comp.category.value:>10}"
            console.print(f"      {cat_str}   {check}  {comp.relative_to_video}")

    # Action keys
    console.print()
    _render_action_keys(console, prompt)


def _render_action_keys(console: Console, prompt: InteractivePrompt) -> None:
    """Render action keys with default highlighted."""
    default = prompt.default_action
    candidates = prompt.candidates
    has_companions = False  # caller can set this later

    keys: list[tuple[str, str, bool]] = []  # (key_label, action_name, is_default)

    if candidates:
        if len(candidates) > 1 and default != PromptAction.ACCEPT:
            # Numbered accept for each candidate
            for i in range(len(candidates)):
                keys.append((str(i + 1), "accept", False))
        else:
            keys.append(("enter", "accept", default == PromptAction.ACCEPT))

    keys.append(("s", "search", default == PromptAction.SEARCH))
    keys.append(("m", "metadata", default == PromptAction.MANUAL))
    keys.append(("x", "skip", default == PromptAction.SKIP))
    keys.append(("q", "quit", False))

    parts = Text("      ")
    for key_label, action_name, is_default in keys:
        if is_default:
            parts.append(f"[enter {action_name}] ", style="bold")
        else:
            parts.append(f"[{key_label}]{action_name} ", style="dim")

    console.print(parts)
```

```python
import sys


def _read_key() -> str:
    """Read a single keypress from stdin."""
    import tty
    import termios
    fd = sys.stdin.fileno()
    old = termios.tcgetattr(fd)
    try:
        tty.setraw(fd)
        return sys.stdin.read(1)
    finally:
        termios.tcsetattr(fd, termios.TCSADRAIN, old)


def edit_companions(
    console: Console,
    companions: list,  # list of CompanionFile
) -> list:
    """Interactive checklist editor for companion files.

    Video files are always selected and locked. Other files can be toggled.
    Returns the list of selected CompanionFile objects.
    """
    from tapes.companions.classifier import Category

    selected = [comp.move_by_default or comp.category == Category.VIDEO for comp in companions]

    while True:
        console.print("\nEdit companion files:")
        for i, comp in enumerate(companions):
            is_video = comp.category == Category.VIDEO
            check = "[bold green]x[/bold green]" if selected[i] else " "
            lock = "  (locked)" if is_video else ""
            console.print(
                f"  [{check}] {i} {comp.relative_to_video}   "
                f"{comp.category.value}{lock}"
            )
        console.print("\nType a number to toggle, Enter to confirm.")

        key = _read_key()
        if key in ("\r", "\n"):
            break
        if key.isdigit():
            idx = int(key)
            if 0 <= idx < len(companions):
                if companions[idx].category != Category.VIDEO:
                    selected[idx] = not selected[idx]

    return [comp for comp, sel in zip(companions, selected) if sel]
```

```python
def read_action(
    prompt: InteractivePrompt,
    *,
    return_choice: bool = False,
) -> PromptAction | tuple[PromptAction, int]:
    """Read a single action key from the user.

    Returns the chosen PromptAction. If return_choice=True and a numbered
    candidate is selected, returns (PromptAction.ACCEPT, candidate_index).
    """
    candidates = prompt.candidates
    has_multiple_ambiguous = (
        len(candidates) > 1 and prompt.default_action != PromptAction.ACCEPT
    )

    while True:
        key = _read_key()

        if key in ("\r", "\n"):
            default = prompt.default_action
            if default is not None:
                if return_choice:
                    return default, 0
                return default
            continue  # no default, ignore enter

        if key == "s":
            return (PromptAction.SEARCH, -1) if return_choice else PromptAction.SEARCH
        if key == "m":
            return (PromptAction.MANUAL, -1) if return_choice else PromptAction.MANUAL
        if key == "x":
            return (PromptAction.SKIP, -1) if return_choice else PromptAction.SKIP
        if key == "q":
            return (PromptAction.QUIT, -1) if return_choice else PromptAction.QUIT
        if key == "a" and candidates:
            return (PromptAction.ACCEPT, -1) if return_choice else PromptAction.ACCEPT

        # Numbered candidate selection
        if key.isdigit() and has_multiple_ambiguous:
            idx = int(key) - 1
            if 0 <= idx < len(candidates):
                if return_choice:
                    return PromptAction.ACCEPT, idx
                return PromptAction.ACCEPT

        # Invalid key - loop again
```
